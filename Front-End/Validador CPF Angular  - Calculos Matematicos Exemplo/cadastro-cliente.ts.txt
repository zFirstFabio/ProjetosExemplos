Abra o seu cadastro-cliente.ts e adicione estas duas coisas dentro da sua classe:

Uma variável para controlar a tela:

TypeScript
  // Adicione junto com as outras variáveis lá no topo da classe
  cpfValido: boolean = true;
A função oficial de cálculo de CPF (pode colar no final do arquivo, perto da função de excluir):

TypeScript
  // Verifica matematicamente se o CPF existe
  verificarCpf(): void {
    let cpf = this.cliente.cpf ? this.cliente.cpf.replace(/\D/g, '') : '';
    
    // Se estiver vazio, não valida ainda (deixa o "required" do HTML cuidar disso)
    if (cpf === '') {
      this.cpfValido = true;
      return;
    }

    // CPFs com tamanho errado ou números repetidos (ex: 111.111.111-11)
    if (cpf.length !== 11 || /^(\d)\1+$/.test(cpf)) {
      this.cpfValido = false;
      return;
    }

    // Cálculo matemático oficial dos dígitos verificadores
    let soma = 0;
    let resto;
    for (let i = 1; i <= 9; i++) soma = soma + parseInt(cpf.substring(i - 1, i)) * (11 - i);
    resto = (soma * 10) % 11;
    if ((resto == 10) || (resto == 11)) resto = 0;
    if (resto != parseInt(cpf.substring(9, 10))) {
      this.cpfValido = false;
      return;
    }

    soma = 0;
    for (let i = 1; i <= 10; i++) soma = soma + parseInt(cpf.substring(i - 1, i)) * (12 - i);
    resto = (soma * 10) % 11;
    if ((resto == 10) || (resto == 11)) resto = 0;
    if (resto != parseInt(cpf.substring(10, 11))) {
      this.cpfValido = false;
      return;
    }

    this.cpfValido = true; // Passou em todos os testes!
  }
